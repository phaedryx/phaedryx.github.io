class Alchemy::OmniauthCallbacksController < Devise::OmniauthCallbacksController
  def facebook
    @user = Alchemy::User.from_omniauth(request.env["omniauth.auth"])

    sign_in_and_redirect @user, :event => :authentication
  end

  def google_oauth2
    @user = Alchemy::User.from_omniauth(request.env["omniauth.auth"])

    sign_in_and_redirect @user, :event => :authentication
  end
end
